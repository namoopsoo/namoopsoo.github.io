lpad is the func to make sure a month is always two digits as an example.#


select 
concat( extract (year from  foo.timestamp)::text, 
lpad (extract (month from  foo.timestamp)::text, 2, '0') ) 

 as yearmonth, 
count(1) 
from foo
where 
group by yearmonth 
order by yearmonth  asc
